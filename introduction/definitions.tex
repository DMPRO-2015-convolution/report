\subsection{Definitions}

\begin{description}
    \item[Processor]
    \hfill\\
    The processor implementation on the FPGA.
    \item[Board]
    \hfill\\
    The printed circuit board (PCB).
    \item[Computer]
    \hfill\\
    The board with soldered components.
    \item[System]
    \hfill\\
    The computer with peripherals, power supply, etc.
\end{description}
